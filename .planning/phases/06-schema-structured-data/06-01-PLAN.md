4. **BreadcrumbList** — on all pages EXCEPT homepage (pageType !== 'home'):
Build breadcrumb items from the slug. Logic:
```typescript
function buildBreadcrumbs(locale: Locale, roSlug?: string, enSlug?: string, title?: string): any[] | null {
  const slug = locale === 'ro' ? (roSlug || '/') : (enSlug || '/en/');
  if (slug === '/' || slug === '/en/') return null; // no breadcrumbs on homepage

  const items: any[] = [];
  const homeName = locale === 'ro' ? 'Acasa' : 'Home';
  const homeUrl = locale === 'ro' ? baseUrl + '/' : baseUrl + '/en/';

  items.push({ '@type': 'ListItem', position: 1, name: homeName, item: homeUrl });

  const segments = slug.split('/').filter(Boolean);
  // Remove 'en' prefix for EN locale
  const pathSegments = locale === 'en' ? segments.slice(1) : segments;

  // Build intermediate crumbs from slugMap
  const slugLabels: Record<string, Record<Locale, string>> = {
    'servicii': { ro: 'Servicii', en: 'Services' },
    'services': { ro: 'Servicii', en: 'Services' },
    'portofoliu': { ro: 'Portofoliu', en: 'Portfolio' },
    'portfolio': { ro: 'Portofoliu', en: 'Portfolio' },
    'pret': { ro: 'Preturi', en: 'Pricing' },
    'pricing': { ro: 'Preturi', en: 'Pricing' },
    'despre': { ro: 'Despre', en: 'About' },
    'about': { ro: 'Despre', en: 'About' },
    'contact': { ro: 'Contact', en: 'Contact' },
  };

  let currentPath = locale === 'ro' ? '' : '/en';
  for (let i = 0; i < pathSegments.length; i++) {
    currentPath += '/' + pathSegments[i];
    const isLast = i === pathSegments.length - 1;
    const label = isLast && title ? title.split(' — ')[0] : (slugLabels[pathSegments[i]]?.[locale] || pathSegments[i]);
    const crumb: any = { '@type': 'ListItem', position: i + 2, name: label };
    if (!isLast) crumb.item = baseUrl + currentPath + '/';
    items.push(crumb);
  }

  return items;
}
```
