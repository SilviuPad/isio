    2. **admin/src/pages/api/email-pdf.ts** — Email PDF via Resend (per D-20):
       ```typescript
       export const prerender = false;

       import type { APIContext } from 'astro';
       import { Resend } from 'resend';

       export async function POST({ request, locals }: APIContext) {
         const { to, filename, pdfBase64, subject } = await request.json();

         if (!to || !filename || !pdfBase64) {
           return new Response(
             JSON.stringify({ error: 'Missing required fields: to, filename, pdfBase64' }),
             { status: 400, headers: { 'Content-Type': 'application/json' } }
           );
         }

         const resend = new Resend(locals.runtime.env.RESEND_API_KEY);

         try {
           const { error } = await resend.emails.send({
             from: 'Isio <onboarding@resend.dev>',
             to,
             subject: subject || `Document from Isio: ${filename}`,
             html: `
               <div style="font-family: sans-serif; padding: 20px;">
                 <h2>Document from Isio</h2>
                 <p>Please find the attached document: <strong>${filename}</strong></p>
                 <p>If you have any questions, please contact us at contact@isio.ro.</p>
                 <hr style="border: none; border-top: 1px solid #e5e7eb; margin: 20px 0;" />
                 <p style="color: #6b7280; font-size: 12px;">Isio — Web Agency | isio.ro</p>
               </div>
             `,
             attachments: [
               {
                 filename,
                 content: pdfBase64,
               },
             ],
           });

           if (error) {
             console.error('Resend error:', error);
             return new Response(
               JSON.stringify({ error: 'Failed to send email' }),
               { status: 500, headers: { 'Content-Type': 'application/json' } }
             );
           }

           return new Response(
             JSON.stringify({ success: true }),
             { status: 200, headers: { 'Content-Type': 'application/json' } }
           );
         } catch (err) {
           console.error('Email exception:', err);
           return new Response(
             JSON.stringify({ error: 'Failed to send email' }),
             { status: 500, headers: { 'Content-Type': 'application/json' } }
           );
         }
       }
       ```

       The Resend `attachments` field accepts `{ filename, content }` where content is a base64 string. The client-side script sends the PDF as base64 from `doc.output('datauristring').split(',')[1]`.
