**src/components/seo/JsonLd.astro:**
Props: `{ locale: Locale; pageType?: 'home' | 'service' | 'pricing' | 'portfolio' | 'about' | 'contact' | 'default'; serviceName?: string }`.

Generates JSON-LD `<script type="application/ld+json">` tags. Always includes WebSite and LocalBusiness schema. Conditionally includes Service schema on service pages.

```astro
---
import type { Locale } from '../../i18n/utils';

interface Props {
  locale: Locale;
  pageType?: 'home' | 'service' | 'pricing' | 'portfolio' | 'about' | 'contact' | 'default';
  serviceName?: string;
}

const { locale, pageType = 'default', serviceName } = Astro.props;
const baseUrl = 'https://isio.ro';

// WebSite schema -- always present
const websiteSchema = {
  '@context': 'https://schema.org',
  '@type': 'WebSite',
  name: 'Isio',
  url: baseUrl,
  inLanguage: locale === 'ro' ? 'ro-RO' : 'en-US',
  description: locale === 'ro'
    ? 'Agenție web specializată în dezvoltare, SEO, accesibilitate și agenți AI.'
    : 'Web agency specializing in development, SEO, accessibility, and AI agents.',
};

// LocalBusiness schema -- always present
const localBusinessSchema = {
  '@context': 'https://schema.org',
  '@type': 'ProfessionalService',
  name: 'Isio',
  url: baseUrl,
  logo: `${baseUrl}/favicon.svg`,
  description: locale === 'ro'
    ? 'Agenție web specializată în dezvoltare website-uri, aplicații web, SEO și accesibilitate.'
    : 'Web agency specializing in website development, web apps, SEO, and accessibility.',
  address: {
    '@type': 'PostalAddress',
    addressCountry: 'RO',
  },
  areaServed: [
    { '@type': 'Country', name: 'Romania' },
    { '@type': 'Country', name: 'Global' },
  ],
  knowsLanguage: ['ro', 'en'],
  priceRange: '€€',
  serviceType: [
    'Web Development',
    'Web Application Development',
    'Search Engine Optimization',
    'Web Accessibility Auditing',
    'AI Agent Implementation',
  ],
};

// Service schema -- only on service pages
const serviceSchema = serviceName ? {
  '@context': 'https://schema.org',
  '@type': 'Service',
  name: serviceName,
  provider: {
    '@type': 'ProfessionalService',
    name: 'Isio',
    url: baseUrl,
  },
  areaServed: {
    '@type': 'Country',
    name: 'Romania',
  },
} : null;

const schemas = [websiteSchema, localBusinessSchema, serviceSchema].filter(Boolean);
---
{schemas.map((schema) => (
  <script type="application/ld+json" set:html={JSON.stringify(schema)} />
))}
```

**src/components/layout/MobileMenu.astro:**
A mobile overlay menu that appears when the hamburger is clicked. Uses a `<details>` element for CSS-only toggle (zero client JS) or minimal inline `<script>` to toggle visibility.

Props: `{ locale: Locale; roSlug: string; enSlug: string }`.

Implementation using a small inline script for toggle:
```astro
---
import LanguageSwitcher from './LanguageSwitcher.astro';
import { localePath } from '../../i18n/utils';
import type { Locale } from '../../i18n/utils';

interface Props {
  locale: Locale;
  roSlug: string;
  enSlug: string;
}

const { locale, roSlug, enSlug } = Astro.props;

const navLabels: Record<Locale, Record<string, string>> = {
  ro: { services: 'Servicii', portfolio: 'Portofoliu', pricing: 'Prețuri', about: 'Despre', contact: 'Contact' },
  en: { services: 'Services', portfolio: 'Portfolio', pricing: 'Pricing', about: 'About', contact: 'Contact' },
};
const labels = navLabels[locale];
const navItems = [
  { label: labels.services, href: localePath('services', locale) },
  { label: labels.portfolio, href: localePath('portfolio', locale) },
  { label: labels.pricing, href: localePath('pricing', locale) },
  { label: labels.about, href: localePath('about', locale) },
  { label: labels.contact, href: localePath('contact', locale) },
];
---
<div id="mobile-menu" class="fixed inset-0 z-50 bg-navy-950/95 backdrop-blur-sm hidden md:hidden">
  <div class="flex flex-col h-full px-6 py-6">
    <div class="flex items-center justify-between">
      <a href={localePath('home', locale)} class="text-xl font-bold text-white font-heading">Isio</a>
      <button id="mobile-menu-close" class="text-white p-2" aria-label="Close menu">
        <svg class="w-6 h-6" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />
        </svg>
      </button>
    </div>
    <nav class="flex flex-col gap-2 mt-12">
      {navItems.map((item) => (
        <a href={item.href} class="text-2xl font-semibold text-white hover:text-electric-400 py-3 font-heading transition-colors">
          {item.label}
        </a>
      ))}
    </nav>
    <div class="mt-auto pb-8">
      <LanguageSwitcher currentLocale={locale} roSlug={roSlug} enSlug={enSlug} />
    </div>
  </div>
</div>
<script>
  const openBtn = document.getElementById('mobile-menu-open');
  const closeBtn = document.getElementById('mobile-menu-close');
  const menu = document.getElementById('mobile-menu');
  if (openBtn && closeBtn && menu) {
    openBtn.addEventListener('click', () => { menu.classList.remove('hidden'); document.body.style.overflow = 'hidden'; });
    closeBtn.addEventListener('click', () => { menu.classList.add('hidden'); document.body.style.overflow = ''; });
    menu.querySelectorAll('a').forEach(link => {
      link.addEventListener('click', () => { menu.classList.add('hidden'); document.body.style.overflow = ''; });
    });
  }
</script>
```
